 27: /**
 28:  * Represents a non user-bound connection to the figo Connect API
 29:  */
 30: class Connection {
 31: 
 32:     private $client_id;
 33:     private $client_secret;
 34:     private $redirect_uri;
 35: 
 36:     /**
 37:      * Constructor
 38:      *
 39:      * @param string the client ID
 40:      * @param string the client secret
 41:      * @param string redirect URI
 42:      */
 43:     public function __construct($client_id, $client_secret, $redirect_uri = null) {
 44:         $this->client_id = $client_id;
 45:         $this->client_secret = $client_secret;
 46:         $this->redirect_uri = $redirect_uri;
 47:     }
 48: 
 49:     /**
 50:      * Helper method for making a OAuth 2.0 request 
 51:      *
 52:      * @param string the URL path on the server
 53:      * @param array this optional associative array will be used as url-encoded POST content.
 54:      * @return array JSON response
 55:      */
 56:     public function query_api($path, array $data = null) {
 57:         $data = is_null($data) ? "" : http_build_query($data);
 58: 
 59:         $headers = array("Authorization"  => "Basic ".base64_encode($this->client_id.":".$this->client_secret),
 60:                          "Content-Type"   => "application/x-www-form-urlencoded",
 61:                          "Content-Length" => strlen($data));
 62: 
 63:         $request = new HttpsRequest();
 64:         return $request->request($path, $data, "POST", $headers);
 65:     }
 66: 
 67:     /**
 68:      * Get the URL a user should open in the web browser to start the login process
 69:      *
 70:      * When the process is completed, the user is redirected to the URL provided to
 71:      * the constructor and passes on an authentication code. This code can be converted
 72:      * into an access token for data access.
 73:      *
 74:      * @param string this string will be passed on through the complete login
 75:      *        process and to the redirect target at the end. It should be used to
 76:      *        validated the authenticity of the call to the redirect URL
 77:      * @param string scope of data access to ask the user for, e.g. <code>accounts=ro</code>
 78:      * @return string the URL to be opened by the user
 79:      */
 80:     public function login_url($state, $scope = null) {
 81:         $data = array("response_type" => "code",
 82:                       "client_id"     => $this->client_id,
 83:                       "state"         => $state);
 84:         if (!is_null($this->redirect_uri)) {
 85:             $data["redirect_uri"] = $this->redirect_uri;
 86:         }
 87:         if (!is_null($scope)) {
 88:             $data["scope"] = $scope;
 89:         }
 90:         return "https://".Config::$API_ENDPOINT."/auth/code?".http_build_query($data);
 91:     }
 92: 
 93:     /**
 94:      * Exchange authorization code or refresh token for access token.
 95:      *
 96:      * @param string either the authorization code received as part of the call to the 
 97:      *        redirect URL at the end of the logon process, or a refresh token
 98:      * @param string scope of data access to ask the user for, e.g. <code>accounts=ro</code>
 99:      * @return array associative array with the keys <code>access_token</code>, <code>refresh_token</code>
100:      *         and <code>expires</code>, as documented in the figo Connect API specification
101:      */
102:     public function obtain_access_token($authorization_code_or_refresh_token, $scope = null) {
103:       // Authorization codes always start with "O" and refresh tokens always start with "R".
104:       if ($authorization_code_or_refresh_token[0] === "O") {
105:         $data = array("grant_type" => "authorization_code", "code" => $authorization_code_or_refresh_token);
106:         if (!is_null($this->redirect_uri)) {
107:             $data["redirect_uri"] = $this->redirect_uri;
108:         }
109:       } elseif ($authorization_code_or_refresh_token[0] === "R") {
110:         $data = array("grant_type" => "refresh_token", "refresh_token" => $authorization_code_or_refresh_token);
111:         if (!is_null($scope)) {
112:             $data["scope"] = $scope;
113:         }
114:       }
115:       return $this->query_api("/auth/token", $data);
116:     }
117: 
118:     /**
119:      * Revoke refresh token or access token.
120:      *
121:      * This action has immediate effect, i.e. you will not be able use that token anymore after this call.
122:      *
123:      * @param string access or refresh token to be revoked
124:      */
125:     public function revoke_token($refresh_token_or_access_token) {
126:       $data = array("token" => $refresh_token_or_access_token);
127:       $this->query_api("/auth/revoke?".http_build_query($data));
128:     }
129: 
130: }
