 28:  * HTTPS request class with certificate authentication and enhanced error handling
 29:  */
 30: class HttpsRequest {
 31: 
 32:     /**
 33:      * Send client request and return server response.
 34:      *
 35:      * @param string the URL path on the server
 36:      * @param string the HTTP body
 37:      * @param string the HTTP method
 38:      * @param array additional HTTP headers
 39:      * @return array JSON response
 40:      */
 41:     public function request($path, $data, $method, array $headers) {
 42:         // Open socket.
 43:         $context = stream_context_create();
 44:         stream_context_set_option($context, "ssl", "verify_peer", true);
 45:         stream_context_set_option($context, "ssl", "capture_peer_cert", true);
 46: 
 47:         $fp = stream_socket_client("tls://".Config::$API_ENDPOINT.":443/", $errno, $errstr, 60, STREAM_CLIENT_CONNECT, $context);
 48:         if (!$fp) {
 49:             throw new Exception("socket_error", $errstr);
 50:         }
 51:         stream_set_timeout($fp, 60);
 52: 
 53:         // Verify fingerprint of server SSL/TLS certificate.
 54:         $options = stream_context_get_options($context);
 55:         if (isset($options["ssl"]) && isset($options["ssl"]["peer_certificate"])) {
 56:             $certificate = $options["ssl"]["peer_certificate"];
 57:             openssl_x509_export($certificate, $certificate);
 58:             $fingerprint = sha1(base64_decode(preg_replace("/-.*/", "", $certificate)));
 59:             $fingerprint = implode(":", str_split(strtoupper($fingerprint), 2));
 60:             if (!in_array($fingerprint, Config::$VALID_FINGERPRINTS)) {
 61:                 fclose($fp);
 62:                 throw new Exception("ssl_error", "SSL/TLS certificate fingerprint mismatch.");
 63:             }
 64:         }
 65: 
 66:         // Setup common HTTP headers.
 67:         $headers["Host"] = Config::$API_ENDPOINT;
 68:         $headers["Accept"] = "application/json";
 69:         $headers["User-Agent"] = "php-figo";
 70:         $headers["Connection"] = "close";
 71: 
 72:         // Send client request.
 73:         $header = "";
 74:         foreach ($headers as $key => $value) {
 75:             $header .= $key.": ".$value."\r\n";
 76:         }
 77:         fwrite($fp, $method." ".$path." HTTP/1.1\r\n".$header."\r\n".$data);
 78: 
 79:         // Read server response.
 80:         $response = stream_get_contents($fp);
 81:         fclose($fp);
 82:         preg_match("/ (\d+)/", $response, $code);
 83:         $code = intval($code[1]);
 84:         $body = substr($response, strpos($response, "\r\n\r\n") + 4);
 85: 
 86:         // Evaluate HTTP response.
 87:         if ($code >= 200 && $code < 300) {
 88:             if (strlen($body) === 0) {
 89:                 return array();
 90:             } else {
 91:                 $obj = json_decode($body, true);
 92:                 if (is_null($obj)) {
 93:                     throw new Exception("json_error", "Cannot decode JSON object.");
 94:                 } else {
 95:                     return $obj;
 96:                 }
 97:              }
 98:         } elseif ($code === 400) {
 99:              $err = json_decode($body, true);
100:              if (is_null($err)) {
101:                  throw new Exception("json_error", "Cannot decode JSON object.");
102:              } else {
103:                  throw new Exception($err["error"], $err["error_description"]);
104:              }
105:         } elseif ($code === 401) {
106:             throw new Exception("unauthorized", "Missing, invalid or expired access token.");
107:         } elseif ($code === 403) {
108:             throw new Exception("forbidden", "Insufficient permission.");
109:         } elseif ($code === 404) {
110:             return null;
111:         } elseif ($code === 405) {
112:             throw new Exception("method_not_allowed", "Unexpected request method.");
113:         } elseif ($code === 503) {
114:             throw new Exception("service_unavailable", "Exceeded rate limit.");
115:         } else {
116:             throw new Exception("internal_server_error", "We are very sorry, but something went wrong.");
117:         }
118:     }
119: }
